### 1. 复利计算器 💰（P0）

**核心价值**：让用户看到坚持储蓄和投资的力量

#### 功能描述

- 输入：初始本金、每月定投、年化收益率、投资年限
- 输出：总资产、总收益、收益率、年度增长曲线
- 可视化：资产增长曲线图（折线图）
- 对比：纯储蓄 vs 投资收益的对比

#### 技术实现

```javascript
/**
 * 复利计算器
 * @param {number} principal - 初始本金
 * @param {number} monthlyContribution - 每月定投金额
 * @param {number} years - 投资年限
 * @param {number} annualRate - 年化收益率（小数，如0.08表示8%）
 * @returns {object} 计算结果
 */
function calculateCompoundInterest(principal, monthlyContribution, years, annualRate) {
  const monthlyRate = annualRate / 12;
  const months = years * 12;
  
  // 本金的复利
  const principalFuture = principal * Math.pow(1 + monthlyRate, months);
  
  // 每月定投的复利（年金终值公式）
  const contributionFuture = monthlyContribution * 
    ((Math.pow(1 + monthlyRate, months) - 1) / monthlyRate);
  
  const totalFuture = principalFuture + contributionFuture;
  const totalContribution = principal + (monthlyContribution * months);
  const totalInterest = totalFuture - totalContribution;
  
  // 计算每年的资产增长（用于可视化）
  const yearlyData = [];
  for (let year = 1; year <= years; year++) {
    const yearMonths = year * 12;
    const yearPrincipalFuture = principal * Math.pow(1 + monthlyRate, yearMonths);
    const yearContributionFuture = monthlyContribution * 
      ((Math.pow(1 + monthlyRate, yearMonths) - 1) / monthlyRate);
    const yearTotal = yearPrincipalFuture + yearContributionFuture;
    
    yearlyData.push({
      year: year,
      totalAssets: Math.round(yearTotal),
      totalContribution: principal + (monthlyContribution * yearMonths),
      totalInterest: Math.round(yearTotal - principal - (monthlyContribution * yearMonths))
    });
  }
  
  return {
    totalFuture: Math.round(totalFuture),
    totalContribution: Math.round(totalContribution),
    totalInterest: Math.round(totalInterest),
    interestRatio: ((totalInterest / totalContribution) * 100).toFixed(1),
    yearlyData: yearlyData
  };
}

// 示例1：每月存20,000元，8%年化，10年后
const example1 = calculateCompoundInterest(0, 20000, 10, 0.08);
// 结果：
// - 总投入：2,400,000元
// - 总收益：1,219,712元
// - 总资产：3,619,712元
// - 收益率：50.8%

// 示例2：初始本金100,000元，每月存20,000元，8%年化，10年后
const example2 = calculateCompoundInterest(100000, 20000, 10, 0.08);
// 结果：
// - 总投入：2,500,000元
// - 总收益：1,448,644元
// - 总资产：3,948,644元
// - 收益率：57.9%
```

### 2. 提前还款计算器 🏠（P1）

**核心价值**：帮助用户科学决策是否提前还贷

#### 功能描述

- 输入：贷款总额、贷款期限、贷款利率、已还期数、提前还款金额
- 输出：节省利息、新还款期限、投资对比
- 对比：提前还款 vs 投资收益（哪个更划算？）

#### 技术实现

```javascript
/**
 * 等额本息贷款计算
 * @param {number} principal - 贷款总额
 * @param {number} annualRate - 年利率（小数）
 * @param {number} months - 贷款期限（月）
 * @returns {object} 还款信息
 */
function calculateLoan(principal, annualRate, months) {
  const monthlyRate = annualRate / 12;
  
  // 等额本息月供公式
  const monthlyPayment = principal * 
    (monthlyRate * Math.pow(1 + monthlyRate, months)) / 
    (Math.pow(1 + monthlyRate, months) - 1);
  
  const totalPayment = monthlyPayment * months;
  const totalInterest = totalPayment - principal;
  
  return {
    monthlyPayment: Math.round(monthlyPayment),
    totalPayment: Math.round(totalPayment),
    totalInterest: Math.round(totalInterest)
  };
}

/**
 * 提前还款计算
 * @param {number} originalPrincipal - 原贷款总额
 * @param {number} annualRate - 年利率（小数）
 * @param {number} totalMonths - 原贷款期限（月）
 * @param {number} paidMonths - 已还期数
 * @param {number} prepayAmount - 提前还款金额
 * @param {string} prepayType - 提前还款类型：'reduce_months'（减少期数）或 'reduce_payment'（减少月供）
 * @returns {object} 提前还款结果
 */
function calculatePrepayment(originalPrincipal, annualRate, totalMonths, paidMonths, prepayAmount, prepayType = 'reduce_months') {
  const monthlyRate = annualRate / 12;
  
  // 原贷款信息
  const originalLoan = calculateLoan(originalPrincipal, annualRate, totalMonths);
  
  // 已还本金计算
  let paidPrincipal = 0;
  for (let month = 1; month <= paidMonths; month++) {
    const interest = (originalPrincipal - paidPrincipal) * monthlyRate;
    const principalPayment = originalLoan.monthlyPayment - interest;
    paidPrincipal += principalPayment;
  }
  
  // 剩余本金
  const remainingPrincipal = originalPrincipal - paidPrincipal;
  
  // 提前还款后剩余本金
  const newPrincipal = remainingPrincipal - prepayAmount;
  
  // 剩余期数
  const remainingMonths = totalMonths - paidMonths;
  
  let result = {
    originalLoan: originalLoan,
    paidMonths: paidMonths,
    remainingPrincipal: Math.round(remainingPrincipal),
    prepayAmount: prepayAmount,
    newPrincipal: Math.round(newPrincipal)
  };
  
  if (prepayType === 'reduce_months') {
    // 减少期数（月供不变）
    // 计算新期数
    const newMonths = Math.ceil(
      Math.log(originalLoan.monthlyPayment / (originalLoan.monthlyPayment - newPrincipal * monthlyRate)) / 
      Math.log(1 + monthlyRate)
    );
    
    const newLoan = calculateLoan(newPrincipal, annualRate, newMonths);
    
    result.type = '减少期数';
    result.newLoan = newLoan;
    result.savedMonths = remainingMonths - newMonths;
    result.savedInterest = originalLoan.totalInterest - 
      (paidMonths * (originalLoan.monthlyPayment - remainingPrincipal * monthlyRate / originalLoan.monthlyPayment)) - 
      newLoan.totalInterest;
  } else {
    // 减少月供（期数不变）
    const newLoan = calculateLoan(newPrincipal, annualRate, remainingMonths);
    
    result.type = '减少月供';
    result.newLoan = newLoan;
    result.savedPayment = originalLoan.monthlyPayment - newLoan.monthlyPayment;
    result.savedInterest = originalLoan.totalInterest - 
      (paidMonths * (originalLoan.monthlyPayment - remainingPrincipal * monthlyRate / originalLoan.monthlyPayment)) - 
      newLoan.totalInterest;
  }
  
  // 对比投资收益
  const investmentReturn = calculateCompoundInterest(prepayAmount, 0, remainingMonths / 12, 0.08);
  result.investmentAlternative = {
    amount: investmentReturn.totalInterest,
    comparison: investmentReturn.totalInterest > result.savedInterest ? '投资更划算' : '提前还款更划算'
  };
  
  return result;
}

// 示例：房贷100万，4.9%利率，30年，已还5年，提前还款20万
const example = calculatePrepayment(1000000, 0.049, 360, 60, 200000, 'reduce_months');
// 结果：
// - 原月供：5,307元
// - 剩余本金：约91万
// - 提前还款后：71万
// - 新期限：约20年（减少5年）
// - 节省利息：约18万
// - 对比投资：20万投资8%年化25年 = 约100万收益（投资更划算）
```

### 3. 退休规划工具 👴（P2）

**核心价值**：帮助用户规划退休所需资金

#### 功能描述

- 输入：当前年龄、退休年龄、当前资产、每月储蓄、预期退休支出
- 输出：退休时总资产、可持续年限、缺口分析
- 建议：需要增加多少储蓄

#### 技术实现

```javascript
/**
 * 退休规划计算
 * @param {number} currentAge - 当前年龄
 * @param {number} retirementAge - 退休年龄
 * @param {number} currentAssets - 当前资产
 * @param {number} monthlySavings - 每月储蓄
 * @param {number} annualReturn - 年化收益率（小数）
 * @param {number} monthlyExpense - 预期退休月支出
 * @param {number} lifeExpectancy - 预期寿命
 * @returns {object} 退休规划结果
 */
function calculateRetirement(currentAge, retirementAge, currentAssets, monthlySavings, annualReturn, monthlyExpense, lifeExpectancy = 85) {
  const yearsToRetirement = retirementAge - currentAge;
  const retirementYears = lifeExpectancy - retirementAge;
  
  // 计算退休时总资产
  const retirementAssets = calculateCompoundInterest(currentAssets, monthlySavings, yearsToRetirement, annualReturn);
  
  // 计算退休后可持续年限
  // 假设退休后资产按保守收益率（3%）增长
  const conservativeReturn = 0.03;
  const monthlyRetirementReturn = conservativeReturn / 12;
  
  let assets = retirementAssets.totalFuture;
  let months = 0;
  
  while (assets > 0 && months < retirementYears * 12) {
    const monthlyIncome = assets * monthlyRetirementReturn;
    assets = assets + monthlyIncome - monthlyExpense;
    months++;
  }
  
  const sustainableYears = Math.floor(months / 12);
  
  // 计算缺口
  const requiredAssets = monthlyExpense * 12 * retirementYears;
  const gap = requiredAssets - retirementAssets.totalFuture;
  
  // 计算需要增加的月储蓄
  let additionalSavings = 0;
  if (gap > 0) {
    // 反推需要的月储蓄
    const targetAssets = requiredAssets;
    const monthlyRate = annualReturn / 12;
    const totalMonths = yearsToRetirement * 12;
    
    // 简化计算：假设初始本金为0
    const requiredMonthlyContribution = targetAssets / 
      ((Math.pow(1 + monthlyRate, totalMonths) - 1) / monthlyRate);
    
    additionalSavings = Math.ceil(requiredMonthlyContribution - monthlySavings);
  }
  
  return {
    retirementAge: retirementAge,
    yearsToRetirement: yearsToRetirement,
    retirementAssets: retirementAssets.totalFuture,
    monthlyExpense: monthlyExpense,
    sustainableYears: sustainableYears,
    retirementYears: retirementYears,
    isSustainable: sustainableYears >= retirementYears,
    gap: Math.round(gap),
    additionalSavingsNeeded: additionalSavings
  };
}

// 示例：30岁，60岁退休，当前资产10万，每月存2万，8%年化，退休月支出1万
const example = calculateRetirement(30, 60, 100000, 20000, 0.08, 10000, 85);
// 结果：
// - 退休时资产：约2,740万
// - 可持续年限：>25年（可持续）
// - 缺口：无
// - 建议：已经足够，可以考虑提高退休生活质量
```

### 4. 大额支出规划 🎯（P2）

**核心价值**：帮助用户规划买房、教育、旅行等大额支出

#### 功能描述

- 输入：目标金额、目标日期、当前储蓄、每月储蓄
- 输出：达成日期、需要增加的月储蓄、投资建议
- 可视化：储蓄进度曲线

#### 技术实现

```javascript
/**
 * 大额支出规划
 * @param {number} targetAmount - 目标金额
 * @param {string} targetDate - 目标日期（YYYY-MM-DD）
 * @param {number} currentSavings - 当前储蓄
 * @param {number} monthlySavings - 每月储蓄
 * @param {number} annualReturn - 年化收益率（小数）
 * @returns {object} 规划结果
 */
function planLargeExpense(targetAmount, targetDate, currentSavings, monthlySavings, annualReturn) {
  const today = new Date();
  const target = new Date(targetDate);
  const monthsToTarget = Math.ceil((target - today) / (1000 * 60 * 60 * 24 * 30));
  
  // 计算目标日期时的总资产
  const futureAssets = calculateCompoundInterest(currentSavings, monthlySavings, monthsToTarget / 12, annualReturn);
  
  // 判断是否足够
  const isAchievable = futureAssets.totalFuture >= targetAmount;
  
  // 计算达成日期
  let achievementDate = null;
  if (isAchievable) {
    // 反推达成日期
    const monthlyRate = annualReturn / 12;
    let assets = currentSavings;
    let month = 0;
    
    while (assets < targetAmount) {
      const monthlyReturn = assets * monthlyRate;
      assets += monthlyReturn + monthlySavings;
      month++;
    }
    
    achievementDate = new Date(today);
    achievementDate.setMonth(achievementDate.getMonth() + month);
  }
  
  // 计算需要增加的月储蓄
  let additionalSavingsNeeded = 0;
  if (!isAchievable) {
    const monthlyRate = annualReturn / 12;
    const requiredMonthlySavings = (targetAmount - currentSavings * Math.pow(1 + monthlyRate, monthsToTarget)) / 
      ((Math.pow(1 + monthlyRate, monthsToTarget) - 1) / monthlyRate);
    
    additionalSavingsNeeded = Math.ceil(requiredMonthlySavings - monthlySavings);
  }
  
  return {
    targetAmount: targetAmount,
    targetDate: targetDate,
    monthsToTarget: monthsToTarget,
    currentSavings: currentSavings,
    monthlySavings: monthlySavings,
    futureAssets: futureAssets.totalFuture,
    isAchievable: isAchievable,
    achievementDate: achievementDate,
    gap: Math.round(targetAmount - futureAssets.totalFuture),
    additionalSavingsNeeded: additionalSavingsNeeded
  };
}

// 示例：3年后买房，首付100万，当前储蓄20万，每月存2万，8%年化
const example = planLargeExpense(1000000, '2029-03-24', 200000, 20000, 0.08);
// 结果：
// - 3年后总资产：约100万
// - 是否足够：是
// - 达成日期：约3年后
```

### 5. 情景模拟 🔮（P2）

**核心价值**：帮助用户理解不同场景下的财务影响

#### 功能描述

- 输入：当前财务状态、假设变化（收入增加/减少、支出变化等）
- 输出：新的财务自由路径、达成时间变化
- 对比：当前 vs 新场景

#### 技术实现

```javascript
/**
 * 情景模拟
 * @param {object} currentScenario - 当前场景
 * @param {object} newScenario - 新场景
 * @returns {object} 对比结果
 */
function simulateScenario(currentScenario, newScenario) {
  // 计算当前场景的财务自由路径
  const currentPath = calculateFinancialFreedomPath(
    currentScenario.netAssets,
    currentScenario.monthlyIncome,
    currentScenario.monthlyExpense,
    currentScenario.annualReturn
  );
  
  // 计算新场景的财务自由路径
  const newPath = calculateFinancialFreedomPath(
    newScenario.netAssets || currentScenario.netAssets,
    newScenario.monthlyIncome || currentScenario.monthlyIncome,
    newScenario.monthlyExpense || currentScenario.monthlyExpense,
    newScenario.annualReturn || currentScenario.annualReturn
  );
  
  return {
    current: currentPath,
    new: newPath,
    changes: {
      netAssets: (newScenario.netAssets || currentScenario.netAssets) - currentScenario.netAssets,
      monthlyIncome: (newScenario.monthlyIncome || currentScenario.monthlyIncome) - currentScenario.monthlyIncome,
      monthlyExpense: (newScenario.monthlyExpense || currentScenario.monthlyExpense) - currentScenario.monthlyExpense,
      monthlySavings: newPath.monthlySavings - currentPath.monthlySavings,
      financialFreedomYears: newPath.yearsToFinancialFreedom - currentPath.yearsToFinancialFreedom
    },
    insight: generateInsight(currentPath, newPath)
  };
}

/**
 * 生成洞察
 */
function generateInsight(currentPath, newPath) {
  const yearsDifference = currentPath.yearsToFinancialFreedom - newPath.yearsToFinancialFreedom;
  
  if (yearsDifference > 0) {
    return `新场景将使财务自由提前${yearsDifference}年达成！`;
  } else if (yearsDifference < 0) {
    return `新场景将使财务自由推迟${Math.abs(yearsDifference)}年达成。`;
  } else {
    return '新场景对财务自由时间没有影响。';
  }
}

// 示例：如果收入增加5,000元
const currentScenario = {
  netAssets: 0,
  monthlyIncome: 30000,
  monthlyExpense: 10000,
  annualReturn: 0.08
};

const newScenario = {
  monthlyIncome: 35000
};

const result = simulateScenario(currentScenario, newScenario);
// 结果：
// - 当前场景：11年达成财务自由
// - 新场景：9年达成财务自由（提前2年）
```
