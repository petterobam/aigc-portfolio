interface User {

### 2.2 账户 (Account)

**资产账户，用于记录各类资产**

interface Account {

interface Goal {
  id: string;                    // 目标ID
  type: GoalType;                // 目标类型
  targetAmount: number;          // 目标金额
  currentAmount: number;         // 当前金额
  monthlyExpense: number;        // 月支出（用于计算）
  targetDate?: Date;             // 目标日期
  status: GoalStatus;            // 状态
  createdAt: Date;
  updatedAt: Date;
}

enum GoalType {
  GUARANTEE = 'guarantee',       // 财务保障
  SECURITY = 'security',         // 财务安全
  FREEDOM = 'freedom';           // 财务自由
}

enum GoalStatus {
  NOT_STARTED = 'not_started',   // 未开始
  IN_PROGRESS = 'in_progress',   // 进行中
  ACHIEVED = 'achieved';         // 已达成
}

// 计算逻辑
// 财务保障目标 = 月支出 × 保障月数（6-12）
// 财务安全目标 = 月支出 × 150
// 财务自由目标 = (月支出 + 梦想月成本) × 150

## 三、计算指标

### 3.1 净资产

```typescript
function calculateNetWorth(accounts: Account[], debts: Debt[]): number {
  const totalAssets = accounts
    .filter(a => a.includeInNetWorth)
    .reduce((sum, a) => sum + a.balance, 0);
  
  const totalDebts = debts
    .filter(d => !d.isPaidOff)
    .reduce((sum, d) => sum + d.remainingAmount, 0);
  
  return totalAssets - totalDebts;
}
```

### 3.2 月现金流

```typescript
function calculateMonthlyCashFlow(
  transactions: Transaction[],
  month: Date
): { income: number; expense: number; balance: number } {
  const monthTransactions = filterByMonth(transactions, month);
  
  const income = monthTransactions
    .filter(t => t.type === 'income')
    .reduce((sum, t) => sum + t.amount, 0);
  
  const expense = monthTransactions
    .filter(t => t.type === 'expense')
    .reduce((sum, t) => sum + t.amount, 0);
  
  return { income, expense, balance: income - expense };
}
```

### 3.3 三阶段进度

```typescript
function calculateGoalProgress(goal: Goal): number {
  return Math.min((goal.currentAmount / goal.targetAmount) * 100, 100);
}
```

### 3.4 关键比率

```typescript
function calculateRatios(
  netWorth: number,
  monthlyIncome: number,
  monthlyExpense: number,
  totalDebts: number,
  totalAssets: number,
  passiveIncome: number
): Ratios {
  return {
    savingsRate: ((monthlyIncome - monthlyExpense) / monthlyIncome) * 100,
    debtRatio: (totalDebts / totalAssets) * 100,
    passiveIncomeRatio: (passiveIncome / monthlyExpense) * 100,
  };
}
```
