## 📝 测试用例示例

### 示例 1：净资产计算测试

```typescript
// tests/unit/netWorth.test.ts
import { describe, it, expect } from 'vitest'

describe('净资产计算', () => {
  it('应该正确计算净资产', () => {
    const assets = 1100000  // 700k + 400k
    const debts = 80000
    const netWorth = assets - debts
    expect(netWorth).toBe(1020000)
  })

  it('应该正确计算储蓄率', () => {
    const monthlyIncome = 30000
    const monthlyExpense = 10000
    const savingsRate = (monthlyIncome - monthlyExpense) / monthlyIncome
    expect(savingsRate).toBeCloseTo(0.667, 2)
  })
})
```

### 示例 2：目标进度测试

```typescript
// tests/unit/goals.test.ts
import { describe, it, expect } from 'vitest'

describe('目标进度计算', () => {
  const netWorth = 1020000

  it('财务保障进度应该是 1700%', () => {
    const target = 60000
    const progress = (netWorth / target) * 100
    expect(progress).toBe(1700)
  })

  it('财务安全进度应该是 68%', () => {
    const target = 1500000
    const progress = Math.min(100, (netWorth / target) * 100)
    expect(Math.round(progress)).toBe(68)
  })

  it('财务自由进度应该是 34%', () => {
    const target = 3000000
    const progress = Math.min(100, (netWorth / target) * 100)
    expect(Math.round(progress)).toBe(34)
  })
})
```

### 示例 3：端到端测试（Playwright）

```typescript
// tests/e2e/goals.spec.ts
import { test, expect } from '@playwright/test'

test.describe('目标页面测试', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5173/#/goals')
    await page.waitForLoadState('networkidle')
  })

  test('应该显示财务保障已完成', async ({ page }) => {
    const securityProgress = await page.locator('.goal-section:first-child .el-progress__text').textContent()
    expect(securityProgress).toContain('1700%')
  })

  test('应该显示正确的净资产', async ({ page }) => {
    const currentAmount = await page.locator('.goal-section:first-child .info-item:nth-child(2) .value').textContent()
    expect(currentAmount).toContain('1,020,000')
  })
})
```

## 🔄 持续集成

### GitHub Actions 配置

```yaml
name: Test

on: [push, pull_request]

jobs:
  test:
    runs-on: macos-latest

    steps:
      - uses: actions/checkout@v3
      - uses: pnpm/action-setup@v2
        with:
          version: 8
      - uses: actions/setup-node@v3
        with:
          node-version: 18
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install

      - name: Run unit tests
        run: pnpm test:unit

      - name: Run e2e tests
        run: pnpm test:e2e

      - name: Upload coverage
        uses: codecov/codecov-action@v3
```
