```typescript
interface IncomeAction {
  id: string
  strategyId: string           // 策略 ID
  title: string                // 行动标题
  description: string          // 行动描述
  priority: 'high' | 'medium' | 'low'
  deadline?: Date              // 截止日期
  status: 'todo' | 'in_progress' | 'completed'

  createdAt?: Date             // 创建时间
}
```
