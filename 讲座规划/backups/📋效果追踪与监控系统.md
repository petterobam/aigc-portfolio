```python
def calculateLearningEffectiveness(learning_data):
    """
    计算学习效果综合得分
    """
    knowledge_score = learning_data['test_scores'] * 0.3
    skill_score = learning_data['skill_assessment'] * 0.4
    willingness_score = learning_data['application_frequency'] * 0.2
    satisfaction_score = learning_data['satisfaction_rating'] * 0.1
    
    overall_score = (knowledge_score + skill_score + 
                    willingness_score + satisfaction_score)
    
    return {
        'knowledge_score': knowledge_score,
        'skill_score': skill_score,
        'willingness_score': willingness_score,
        'satisfaction_score': satisfaction_score,
        'overall_score': overall_score
    }
```

```python
        'efficiency_score': efficiency_score,
        'quality_score': quality_score,

        'innovation_score': innovation_score,
```

#### 3. ROI计算
```python
def calculateROI(cost_data, benefit_data):
    """
    计算投资回报率
    """
    total_cost = sum(cost_data.values())
    total_benefit = sum(benefit_data.values())
    
    roi = ((total_benefit - total_cost) / total_cost) * 100
    payback_period = total_cost / (total_benefit / 12)  # 月数
    
    return {
        'total_cost': total_cost,
        'total_benefit': total_benefit,
        'roi_percentage': roi,
        'payback_period_months': payback_period,
        'roi_status': 'excellent' if roi > 100 else 'good' if roi > 50 else 'needs_improvement'
    }
```

```python
def prioritizeOptimizations(suggestions):
    """
    根据影响范围、紧急程度、实施难度优化建议优先级
    """
    priority_matrix = {
        '高影响-高紧急': 1,
        '高影响-中紧急': 2,
        '高影响-低紧急': 3,
        '中影响-高紧急': 4,
        '中影响-中紧急': 5,
        '中影响-低紧急': 6,
        '低影响-高紧急': 7,
        '低影响-中紧急': 8,
        '低影响-低紧急': 9
    }
    
    for suggestion in suggestions:
        impact = assessImpact(suggestion)
        urgency = assessUrgency(suggestion)
        difficulty = assessDifficulty(suggestion)
        
        score = priority_matrix[f"{impact}-{urgency}"]
        suggestion['priority_score'] = score
        suggestion['implementation_difficulty'] = difficulty
        suggestion['roi_estimate'] = estimateROI(suggestion)
    
    return sorted(suggestions, key=lambda x: x['priority_score'])
```
