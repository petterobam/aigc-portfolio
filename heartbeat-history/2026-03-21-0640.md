async function readPublishContent(storyId, basePath) {
  const possiblePaths = [
    path.join(basePath, '📤 发布内容', `${storyId}号故事-完整内容.txt`),
    path.join(basePath, '📤 发布内容', `${storyId}-完整内容.txt`),
    path.join(basePath, `stories/归档故事集/${storyId}`, `${storyId}号故事-完整内容.txt`),
  ];

  // 查找并读取发布内容文件
  for (const p of possiblePaths) {
    try {
      await fs.access(p);
      content = await fs.readFile(p, 'utf-8');
      contentPath = p;
      break;
    } catch {
      continue;
    }
  }

  return content;
}
```

**功能**:

async function fillContent(page, content) {
  const contentEditor = await page.waitForSelector(selectors.contentEditor, { timeout: 10000 });

  // 清空编辑器
  await contentEditor.click();
  await page.keyboard.down('Meta');
  await page.keyboard.press('a');
  await page.keyboard.up('Meta');
  await page.waitForTimeout(500);
  await page.keyboard.press('Backspace');
  await page.waitForTimeout(500);

  // 粘贴内容
  await contentEditor.click();
  await page.waitForTimeout(500);

  // 直接设置编辑器的文本内容（比粘贴更稳定）
  await page.evaluate((text) => {
    const editor = document.querySelector('.ProseMirror');
    if (editor) {
      editor.textContent = text;
    }
  }, content);

  await page.waitForTimeout(2000);
}
```

**功能**:

#### 8. 提交发布

```javascript
async function submitPublish(page) {
  // 查找提交按钮
  let submitButton = null;
  const possibleSelectors = [
    selectors.submitButton,
    selectors.nextStepButton,
    'button:has-text("提交审核")',
    'button:has-text("发布")',
    'button:has-text("下一步")',
  ];

  for (const selector of possibleSelectors) {
    try {
      submitButton = await page.waitForSelector(selector, { timeout: 3000 });
      if (submitButton) {
        break;
      }
    } catch {
      continue;
    }
  }

  if (!submitButton) {
    throw new Error('未找到提交按钮');
  }

  // 获取按钮文本
  const buttonText = await submitButton.textContent();
  logInfo(`找到按钮: "${buttonText.trim()}"`);

  // 点击按钮
  await submitButton.click();

  // 等待提交结果
  await page.waitForTimeout(3000);

  return page.url();
}
```

**功能**:
- 自动查找并点击提交按钮
- 支持多种按钮文本（"提交审核"、"发布"、"下一步"）
- 显示最终URL
