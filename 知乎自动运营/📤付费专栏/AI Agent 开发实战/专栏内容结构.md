```python
class LongTermMemory:
    def __init__(self, vector_db, embedding_model):
        self.vector_db = vector_db
        self.embedding_model = embedding_model

    def store(self, text, metadata=None):
        # 1. 向量化文本
        vector = self.embedding_model.encode(text)

        # 2. 存储到向量数据库
        self.vector_db.add(vector, text, metadata)

    def retrieve(self, query, top_k=5):
        # 1. 向量化查询
        query_vector = self.embedding_model.encode(query)

        # 2. 相似度检索
        results = self.vector_db.search(query_vector, top_k=top_k)

        # 3. 返回结果
        return results

    def update(self, text_id, new_text):
        # 1. 删除旧向量
        self.vector_db.delete(text_id)

        # 2. 添加新向量
        self.store(new_text, id=text_id)
```

```python
class ProductionAgent:
    def __init__(self, llm, tools, metrics):
        self.llm = llm
        self.tools = tools
        self.metrics = metrics  # 监控指标

    def run(self, user_input):
        # 1. 记录请求开始
        start_time = time.time()

        try:
            # 2. 执行任务
            result = self._execute_task(user_input)

            # 3. 记录成功指标
            self.metrics.record_success(
                duration=time.time() - start_time,

                duration=time.time() - start_time,
                error_type=type(e).__name__
            )
            raise
```
