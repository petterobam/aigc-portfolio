# 示例1：基础 Prompt 设计模板
class PromptTemplate:
    def __init__(self, task_description, context="", examples=None, output_format=""):
        self.task_description = task_description
        self.context = context
        self.examples = examples or []
        self.output_format = output_format

    def build(self, **kwargs):
        prompt = f"""# 任务描述
{self.task_description}

# 上下文
{self.context}

# 示例
"""
        for i, example in enumerate(self.examples, 1):
            prompt += f"\n## 示例 {i}\n"
            prompt += f"输入: {example['input']}\n"
            prompt += f"输出: {example['output']}\n"

        prompt += f"\n# 输出格式\n{self.output_format}\n"
        prompt += f"\n# 当前任务\n输入: {kwargs.get('input', '')}\n输出: "

        return prompt

# 使用示例
prompt = PromptTemplate(
    task_description="将英文文本翻译为中文",
    context="保持原文的语气和风格",
    examples=[
        {"input": "Hello, how are you?", "output": "你好，你好吗？"},
        {"input": "Good morning!", "output": "早上好！"}
    ],
    output_format="仅输出翻译结果"
)

# 示例3：自动选择最佳 Few-shot 示例
class FewShotSelector:
    def __init__(self, embedding_model):
        self.embedding_model = embedding_model

    def select_by_similarity(self, candidate_examples, input_text, top_k=3):
        # 1. 向量化输入和示例
        input_embedding = self.embedding_model.encode(input_text)
        example_embeddings = [
            self.embedding_model.encode(ex['input'])
            for ex in candidate_examples
        ]

        # 2. 计算相似度
        similarities = [
            self._cosine_similarity(input_embedding, ex_emb)
            for ex_emb in example_embeddings
        ]

        # 3. 选择最相似的 top_k 个示例
        top_indices = np.argsort(similarities)[-top_k:][::-1]
        selected_examples = [candidate_examples[i] for i in top_indices]

        return selected_examples

    def select_by_diversity(self, candidate_examples, top_k=3):
        # 1. 向量化所有示例
        example_embeddings = [
            self.embedding_model.encode(ex['input'])
            for ex in candidate_examples
        ]

        # 2. 使用 K-Means 聚类
        kmeans = KMeans(n_clusters=top_k)
        cluster_labels = kmeans.fit_predict(example_embeddings)

        # 3. 从每个聚类中选择最接近中心的示例
        selected_examples = []
        for cluster_id in range(top_k):
            cluster_indices = np.where(cluster_labels == cluster_id)[0]
            center = kmeans.cluster_centers_[cluster_id]

            # 找到最接近中心的示例
            best_index = cluster_indices[
                np.argmin([
                    self._cosine_similarity(center, example_embeddings[i])
                    for i in cluster_indices
                ])
            ]
            selected_examples.append(candidate_examples[best_index])

        return selected_examples

    def _cosine_similarity(self, vec1, vec2):
        return np.dot(vec1, vec2) / (np.linalg.norm(vec1) * np.linalg.norm(vec2))

# 使用示例
selector = FewShotSelector(embedding_model="sentence-transformers/all-MiniLM-L6-v2")

examples = [
    {"input": "这个产品很棒，我很喜欢", "label": "正面"},
    {"input": "服务太差了，再也不来", "label": "负面"},
    {"input": "一般般，没什么特别", "label": "中性"},
    {"input": "非常满意，强烈推荐", "label": "正面"},
    {"input": "态度恶劣，体验极差", "label": "负面"}
]

# 选择与当前输入最相似的 3 个示例

# 示例4：结构化 Prompt 模板（JSON 格式）
class JSONPromptTemplate:
    def __init__(self, task_description, output_schema, examples=None):
        self.task_description = task_description
        self.output_schema = output_schema
        self.examples = examples or []

    def build(self, input_text):
        prompt = f"""# 任务
{self.task_description}

# 输出格式要求
请严格按照以下 JSON 格式输出：
```json
{json.dumps(self.output_schema, indent=2, ensure_ascii=False)}
```

# 注意事项
- 必须输出合法的 JSON 格式
- 必须包含所有要求的字段
- 字段值必须符合指定的类型和约束

"""

        if self.examples:
            prompt += "# 示例\n\n"
            for i, example in enumerate(self.examples, 1):
                prompt += f"## 示例 {i}\n"
                prompt += f"输入: {example['input']}\n"
                prompt += f"输出:\n```json\n"
                prompt += json.dumps(example['output'], indent=2, ensure_ascii=False)
                prompt += "\n```\n\n"

        prompt += f"# 当前任务\n输入: {input_text}\n输出:\n```json\n"

        return prompt

    def validate_output(self, output_json):
        # 1. 尝试解析 JSON
        try:
            data = json.loads(output_json)
        except json.JSONDecodeError as e:
            return False, f"JSON 格式错误: {e}"

        # 2. 验证字段完整性
        for field_name, field_spec in self.output_schema.items():
            if field_name not in data:
                return False, f"缺少字段: {field_name}"

        # 3. 验证字段类型
        for field_name, field_spec in self.output_schema.items():
            expected_type = field_spec.get('type')
            actual_value = data[field_name]

            if expected_type == 'string' and not isinstance(actual_value, str):
                return False, f"字段 {field_name} 应该是字符串"
            elif expected_type == 'number' and not isinstance(actual_value, (int, float)):
                return False, f"字段 {field_name} 应该是数字"
            elif expected_type == 'array' and not isinstance(actual_value, list):
                return False, f"字段 {field_name} 应该是数组"
            elif expected_type == 'object' and not isinstance(actual_value, dict):
                return False, f"字段 {field_name} 应该是对象"

        return True, "验证通过"

# 使用示例

    output_schema={

        "brand": {"type": "string", "description": "品牌"},

        {
            "input": "Apple iPhone 15 Pro Max 售价 9999 元，支持 5G 和 Face ID",
            "output": {
                "product_name": "iPhone 15 Pro Max",

# 示例5：自动化 Prompt 优化工具
class PromptOptimizer:
    def __init__(self, llm, evaluator):
        self.llm = llm
        self.evaluator = evaluator

    def optimize_by_iteration(self, initial_prompt, test_cases, max_iterations=5):
        current_prompt = initial_prompt
        best_score = self.evaluator.evaluate(current_prompt, test_cases)

        for iteration in range(max_iterations):
            print(f"迭代 {iteration + 1}/{max_iterations}")

            # 1. 生成优化建议
            suggestions = self._generate_suggestions(current_prompt, test_cases, best_score)
            print("优化建议:", suggestions)

            # 2. 应用优化建议
            new_prompt = self._apply_suggestions(current_prompt, suggestions)

            # 3. 评估新 Prompt
            new_score = self.evaluator.evaluate(new_prompt, test_cases)
            print(f"新 Prompt 评分: {new_score:.2f}")

            # 4. 如果效果更好，则更新
            if new_score > best_score:
                best_score = new_score
                current_prompt = new_prompt
                print("✓ 优化成功，更新 Prompt")
            else:
                print("✗ 优化失败，保持原 Prompt")

            print()

        return current_prompt, best_score

    def optimize_by_ab_testing(self, prompt_a, prompt_b, test_cases, alpha=0.05):
        # 1. 评估两个 Prompt
        scores_a = self.evaluator.evaluate_detailed(prompt_a, test_cases)
        scores_b = self.evaluator.evaluate_detailed(prompt_b, test_cases)

        # 2. 统计分析
        mean_a = np.mean(scores_a)
        mean_b = np.mean(scores_b)
        std_a = np.std(scores_a)
        std_b = np.std(scores_b)

        # 3. T 检验
        from scipy import stats
        t_stat, p_value = stats.ttest_ind(scores_a, scores_b)

        # 4. 判断显著性
        is_significant = p_value < alpha

        result = {
            "mean_a": mean_a,
            "mean_b": mean_b,
            "std_a": std_a,
            "std_b": std_b,
            "t_stat": t_stat,
            "p_value": p_value,
            "is_significant": is_significant,
            "better_prompt": "A" if mean_a > mean_b else "B"
        }

        return result

    def _generate_suggestions(self, prompt, test_cases, current_score):
        # 使用 LLM 生成优化建议
        suggestion_prompt = f"""请分析以下 Prompt，并给出优化建议，目标是提升准确率。

当前 Prompt:
{prompt}

当前准确率: {current_score:.2f}

优化建议（每条建议一行）:
"""

        self.safety_rules = safety_rules

        prompt = f"""# 任务
{self.task_description}

# 安全规则
"""

        for rule in self.safety_rules:
            prompt += f"- {rule}\n"

        prompt += "\n# 输入\n"
        prompt += f"用户输入: {input_text}\n"

        if user_context:
            prompt += f"用户上下文: {user_context}\n"

- 如果输入违反安全规则，拒绝处理并说明原因
- 不输出任何敏感、有害、虚假的信息

"""

        # 1. 检查是否包含敏感信息
        sensitive_keywords = ["密码", "信用卡", "身份证", "电话号码", "邮箱"]
        for keyword in sensitive_keywords:
            if keyword in output:
                return False, f"输出包含敏感信息: {keyword}"

                violations.append(rule)

        if violations:
            return False, f"输出违反安全规则: {', '.join(violations)}"

        return True, "验证通过"

# 示例9：完整的 Prompt 优化工具
class PromptOptimizationTool:
    def __init__(self, llm, vector_db, evaluator):
        self.llm = llm
        self.vector_db = vector_db
        self.evaluator = evaluator
        self.templates = {}

    def create_template(self, name, prompt, metadata=None):
        """创建新的 Prompt 模板"""
        template_id = str(uuid.uuid4())

        # 存储模板
        template = {
            "id": template_id,
            "name": name,
            "prompt": prompt,
            "metadata": metadata or {},
            "created_at": datetime.now(),
            "version": 1
        }

        self.templates[template_id] = template

        # 向量化并存储到向量数据库
        embedding = self.vector_db.encode(prompt)
        self.vector_db.add(
            id=template_id,
            vector=embedding,
            metadata={"name": name, "type": "prompt_template"}
        )

        print(f"✓ 模板已创建: {name} (ID: {template_id})")
        return template_id

    def evaluate_template(self, template_id, test_cases):
        """评估 Prompt 模板"""
        if template_id not in self.templates:
            raise ValueError(f"模板不存在: {template_id}")

        prompt_template = self.templates[template_id]

        # 评估
        results = self.evaluator.evaluate(
            prompt_template["prompt"],
            test_cases
        )

        # 记录评估结果
        evaluation = {
            "template_id": template_id,
            "results": results,
            "evaluated_at": datetime.now()
        }

        self.templates[template_id]["last_evaluation"] = evaluation

        print(f"✓ 评估完成: {prompt_template['name']}")
        print(f"  准确率: {results['accuracy']:.2f}")
        print(f"  一致性: {results['consistency']:.2f}")

        return results

    def optimize_template(self, template_id, test_cases, max_iterations=3):
        """优化 Prompt 模板"""
        if template_id not in self.templates:
            raise ValueError(f"模板不存在: {template_id}")

        template = self.templates[template_id]
        current_prompt = template["prompt"]
        best_score = 0

        print(f"开始优化模板: {template['name']}")
        print(f"初始 Prompt:\n{current_prompt}\n")

        for iteration in range(max_iterations):
            print(f"迭代 {iteration + 1}/{max_iterations}")

            # 1. 生成优化建议
            suggestions = self._generate_optimization_suggestions(
                current_prompt, test_cases
            )

            # 2. 应用优化建议
            new_prompt = self._apply_optimization_suggestions(
                current_prompt, suggestions
            )

            # 3. 评估新 Prompt
            results = self.evaluator.evaluate(new_prompt, test_cases)
            score = results["accuracy"]

            print(f"  新 Prompt 评分: {score:.2f}")

            # 4. 如果效果更好，更新
            if score > best_score:
                best_score = score
                current_prompt = new_prompt
                print("  ✓ 优化成功")
            else:
                print("  ✗ 保持原 Prompt")

            print()

        # 保存优化后的 Prompt（新版本）
        template["version"] += 1
        template["prompt"] = current_prompt
        template["optimized_at"] = datetime.now()
        template["best_score"] = best_score

        print(f"✓ 优化完成: {template['name']}")
        print(f"最终评分: {best_score:.2f}")

        return current_prompt, best_score

    def search_templates(self, query, top_k=5):
        """搜索相似的 Prompt 模板"""
        query_embedding = self.vector_db.encode(query)
        results = self.vector_db.search(query_embedding, top_k=top_k)

        return [
            {
                "id": r["id"],
                "name": r["metadata"]["name"],
                "prompt": self.templates[r["id"]]["prompt"]
            }
            for r in results
        ]

    def _generate_optimization_suggestions(self, prompt, test_cases):
        """生成优化建议"""
        suggestion_prompt = f"""请分析以下 Prompt，并给出优化建议。

Prompt:
{prompt}

优化建议（每条一行）:
"""
        suggestions = self.llm.generate(suggestion_prompt).strip().split('\n')
        return [s.strip() for s in suggestions if s.strip()]

    def _apply_optimization_suggestions(self, prompt, suggestions):
        """应用优化建议"""
        apply_prompt = f"""请根据以下优化建议，改进这个 Prompt。

原始 Prompt:
{prompt}

优化建议:
"""

        for i, suggestion in enumerate(suggestions, 1):
            apply_prompt += f"{i}. {suggestion}\n"

        apply_prompt += "\n改进后的 Prompt:"

        return self.llm.generate(apply_prompt).strip()
