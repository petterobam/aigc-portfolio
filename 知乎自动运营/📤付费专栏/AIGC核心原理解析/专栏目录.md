    # RoPE 旋转位置编码
    x1 = x[..., 0::2]
    x2 = x[..., 1::2]
    x1_new = x1 * cos - x2 * sin
    x2_new = x1 * sin + x2 * cos
    return torch.stack([x1_new, x2_new], dim=-1).flatten(-2)

def ddim_sample(model, x_t, t, timestep_schedule):
    alpha_bar_t = get_alpha_bar(t)
    alpha_bar_t_prev = get_alpha_bar(timestep_schedule)
    
    pred_x0 = model.predict_x0(x_t, t)
    
    x_t_prev = (
        alpha_bar_t_prev.sqrt() * pred_x0 +
        (1 - alpha_bar_t_prev).sqrt() * (x_t - alpha_bar_t.sqrt() * pred_x0) / (1 - alpha_bar_t).sqrt()
    )
    
    return x_t_prev

# CLIP 对比学习损失
def contrastive_loss(image_features, text_features, temperature=0.07):
    # 归一化
    image_features = F.normalize(image_features, dim=-1)
    text_features = F.normalize(text_features, dim=-1)
    
    # 计算相似度矩阵
    logits = (image_features @ text_features.T) / temperature
    
    # 对比学习损失
    labels = torch.arange(len(logits))
    loss_i = F.cross_entropy(logits, labels)
    loss_t = F.cross_entropy(logits.T, labels)
    loss = (loss_i + loss_t) / 2
    
    return loss
```

class MoELayer(nn.Module):
    def __init__(self, d_model, n_experts, top_k=2):
        super().__init__()
        self.router = nn.Linear(d_model, n_experts)
        self.experts = nn.ModuleList([
            Expert(d_model) for _ in range(n_experts)
        ])
        self.top_k = top_k
    
    def forward(self, x):
        # 路由决策
        router_logits = self.router(x)
        top_k_experts = torch.topk(router_logits, self.top_k)
        
        # Expert 计算
        outputs = []
        for i, expert in enumerate(self.experts):
            mask = (top_k_experts.indices == i)
            if mask.any():
                expert_output = expert(x[mask])
                outputs.append((mask, expert_output))
        
        # 聚合输出
        output = aggregate_expert_outputs(outputs, top_k_experts.values)
        return output

# LoRA 实现
class LoRALayer(nn.Module):
    def __init__(self, in_features, out_features, rank=8):
        super().__init__()
        self.lora_A = nn.Parameter(torch.randn(in_features, rank))
        self.lora_B = nn.Parameter(torch.zeros(rank, out_features))
        self.scaling = 1.0 / rank
    
    def forward(self, x):
        # 原始权重 + LoRA 权重
        lora_output = x @ (self.lora_A @ self.lora_B) * self.scaling
        return lora_output
```
