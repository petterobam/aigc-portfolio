## 一、大模型部署挑战（10%）

### 1.1 计算成本爆炸

**问题**：
- 7B 参数模型需要 14GB 显存（FP16）
- 70B 参数模型需要 140GB 显存
- 推理成本：7B 模型单次推理约 $0.001

**根源**：
- 模型参数量呈指数增长
- 注意力机制计算复杂度 O(n²)
- 激活值内存占用大

### 1.2 延迟敏感

**问题**：
- 云端推理延迟：200-500ms（网络延迟 + 推理时间）
- 边缘设备推理延迟：10-50ms
- 用户期望：<100ms

**场景**：
- 实时对话：要求 <500ms
- 实时翻译：要求 <300ms
- 自动驾驶：要求 <100ms

### 1.3 硬件多样性

**云端硬件**：
- NVIDIA A100（80GB 显存，312 TFLOPS）
- NVIDIA H100（80GB 显存，1,000 TFLOPS）
- AWS Inf2（Inferentia 2，AI 专用芯片）

**边缘硬件**：
- 高端手机（16GB RAM，Apple Neural Engine）
- 中端 IoT（4GB RAM，NPU）
- 嵌入式设备（512MB RAM，无 GPU）

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

app = FastAPI()

# 加载模型（支持量化）
model = AutoModelForCausalLM.from_pretrained(
    "meta-llama/Llama-2-7b-hf",
    device_map="auto",
    torch_dtype=torch.float16,
    load_in_8bit=True  # 8 位量化
)
tokenizer = AutoTokenizer.from_pretrained("meta-llama/Llama-2-7b-hf")

class PromptRequest(BaseModel):
    prompt: str
    max_length: int = 256
    temperature: float = 0.7

@app.post("/generate")
async def generate(request: PromptRequest):
    try:
        inputs = tokenizer(request.prompt, return_tensors="pt")
        inputs = {k: v.to(model.device) for k, v in inputs.items()}

        with torch.no_grad():
            outputs = model.generate(
                **inputs,
                max_length=request.max_length,
                temperature=request.temperature,
                do_sample=True
            )

        result = tokenizer.decode(outputs[0], skip_special_tokens=True)
        return {"result": result}

    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

```python
class KVCacheModel:
    def __init__(self, model):
        self.model = model
        self.past_key_values = None

    def generate_with_cache(self, input_ids, max_new_tokens=100):
        generated = input_ids

        for _ in range(max_new_tokens):
            # 使用缓存的 past_key_values
            outputs = self.model(
                input_ids,
                past_key_values=self.past_key_values,
                use_cache=True
            )

            logits = outputs.logits[:, -1, :]
            next_token = torch.argmax(logits, dim=-1).unsqueeze(1)

            generated = torch.cat([generated, next_token], dim=-1)
            self.past_key_values = outputs.past_key_values

            # 停止条件
            if next_token.item() == tokenizer.eos_token_id:
                break

        return generated

# 使用示例
kv_model = KVCacheModel(model)
input_ids = tokenizer("Hello", return_tensors="pt")["input_ids"].to(model.device)
output_ids = kv_model.generate_with_cache(input_ids, max_new_tokens=50)
output = tokenizer.decode(output_ids[0], skip_special_tokens=True)
```

**优化3：连续批处理（Continuous Batching）**

```python
from collections import defaultdict
import asyncio

class ContinuousBatchScheduler:
    def __init__(self, model, tokenizer, batch_size=4):
        self.model = model
        self.tokenizer = tokenizer
        self.batch_size = batch_size
        self.active_requests = {}

    async def add_request(self, request_id, prompt, max_tokens=100):
        self.active_requests[request_id] = {
            "prompt": prompt,
            "max_tokens": max_tokens,
            "generated_tokens": [],
            "finished": False
        }

    async def process_batch(self):
        while any(not r["finished"] for r in self.active_requests.values()):
            # 准备批量输入
            batch_inputs = []
            batch_request_ids = []

            for req_id, req in self.active_requests.items():
                if not req["finished"] and len(req["generated_tokens"]) < req["max_tokens"]:
                    if len(req["generated_tokens"]) == 0:
                        # 初始输入
                        input_ids = self.tokenizer(req["prompt"], return_tensors="pt")["input_ids"]
                    else:
                        # 生成下一个 token
                        input_ids = torch.tensor([req["generated_tokens"][-1]])

                    batch_inputs.append(input_ids)
                    batch_request_ids.append(req_id)

            if not batch_inputs:
                break

            # 批量推理
            batch_tensor = torch.cat(batch_inputs, dim=0).to(self.model.device)
            with torch.no_grad():
                outputs = self.model(batch_tensor)

            # 处理输出
            logits = outputs.logits[:, -1, :]
            next_tokens = torch.argmax(logits, dim=-1)

            for req_id, token in zip(batch_request_ids, next_tokens):
                self.active_requests[req_id]["generated_tokens"].append(token.item())

                # 检查完成
                if token.item() == self.tokenizer.eos_token_id:
                    self.active_requests[req_id]["finished"] = True

            await asyncio.sleep(0.001)  # 避免 CPU 占用过高

    def get_results(self):
        results = {}
        for req_id, req in self.active_requests.items():
            if req["finished"]:
                results[req_id] = self.tokenizer.decode(req["generated_tokens"])
        return results

# 使用示例
scheduler = ContinuousBatchScheduler(model, tokenizer, batch_size=4)

async def handle_requests():
    await scheduler.add_request("req1", "Hello", max_tokens=50)
    await scheduler.add_request("req2", "How are you?", max_tokens=50)

    await scheduler.process_batch()
    results = scheduler.get_results()
    return results
```

### 2.3 成本优化

**成本估算模型**：

```python
def estimate_cost(model_params, requests_per_day, avg_tokens, gpu_cost_per_hour):
    """
    估算云端部署成本

    Args:
        model_params: 模型参数量（如 7B）
        requests_per_day: 每日请求数
        avg_tokens: 平均生成 token 数
        gpu_cost_per_hour: GPU 每小时成本（如 $2.5）

    Returns:
        日成本、月成本、单次推理成本
    """
    # 单次推理时间（估算）
    tokens_per_second = 50  # A100 推理速度
    inference_time_per_request = avg_tokens / tokens_per_second  # 秒

    # 每日推理时间
    daily_inference_hours = (requests_per_day * inference_time_per_request) / 3600

    # 成本计算
    daily_cost = daily_inference_hours * gpu_cost_per_hour
    monthly_cost = daily_cost * 30
    cost_per_request = daily_cost / requests_per_day

    return {
        "daily_cost": daily_cost,
        "monthly_cost": monthly_cost,
        "cost_per_request": cost_per_request,
        "daily_inference_hours": daily_inference_hours
    }

# 使用示例
costs = estimate_cost(
    model_params=7e9,
    requests_per_day=10000,
    avg_tokens=100,
    gpu_cost_per_hour=2.5
)

print(f"日成本: ${costs['daily_cost']:.2f}")
print(f"月成本: ${costs['monthly_cost']:.2f}")
print(f"单次推理成本: ${costs['cost_per_request']:.4f}")
```

```python
import torch.nn.utils.prune as prune

def prune_model(model, pruning_ratio=0.3):
    """
    剪枝模型：移除不重要的权重
    """
    for name, module in model.named_modules():
        if isinstance(module, torch.nn.Linear):
            # L1 非结构化剪枝
            prune.l1_unstructured(module, name='weight', amount=pruning_ratio)

            # 移除 mask，永久剪枝
            prune.remove(module, 'weight')

    return model

# 使用示例
```

**压缩3：知识蒸馏（Knowledge Distillation）**

```python
def distill_model(teacher_model, student_model, train_data, epochs=3):
    """
    知识蒸馏：用大模型（teacher）训练小模型（student）
    """
    optimizer = torch.optim.AdamW(student_model.parameters(), lr=1e-5)
    loss_fn = torch.nn.KLDivLoss(reduction="batchmean")

    student_model.train()
    teacher_model.eval()

    for epoch in range(epochs):
        total_loss = 0
        for batch in train_data:
            inputs = batch["input_ids"].to(student_model.device)

            # Teacher 输出（软标签）
            with torch.no_grad():
                teacher_logits = teacher_model(inputs).logits

            # Student 输出
            student_logits = student_model(inputs).logits

            # 蒸馏损失
            loss = loss_fn(
                torch.log_softmax(student_logits / 2.0, dim=-1),
                torch.softmax(teacher_logits / 2.0, dim=-1)
            )

            # 反向传播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            total_loss += loss.item()

        print(f"Epoch {epoch+1}, Loss: {total_loss / len(train_data):.4f}")

    return student_model

# 使用示例
teacher_model = AutoModelForCausalLM.from_pretrained("meta-llama/Llama-2-7b-hf")
student_model = AutoModelForCausalLM.from_pretrained("distilgpt2")  # 小模型

distilled_model = distill_model(teacher_model, student_model, train_data, epochs=3)
```

```python
import torch
import torch.nn as nn

# 定义极小模型（用于嵌入式设备）
class MicroLM(nn.Module):
    def __init__(self, vocab_size=50257, d_model=128, n_heads=4):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.attention = nn.MultiheadAttention(d_model, n_heads)
        self.ffn = nn.Sequential(
            nn.Linear(d_model, d_model * 4),
            nn.ReLU(),
            nn.Linear(d_model * 4, d_model)
        )
        self.output = nn.Linear(d_model, vocab_size)

    def forward(self, x):
        x = self.embedding(x)
        x = x.transpose(0, 1)
        attn_out, _ = self.attention(x, x, x)
        x = attn_out.transpose(0, 1) + x  # 残差连接
        x = x + self.ffn(x)  # 残差连接
        return self.output(x)
```

```python
def decide_deployment_strategy(model_size, requests_per_day, latency_requirement, privacy_sensitivity):
    """
    部署策略决策树

    Args:
        model_size: 模型参数量（B）
        requests_per_day: 每日请求数
        latency_requirement: 延迟要求（ms）
        privacy_sensitivity: 隐私敏感度（1-5，5=最敏感）

    Returns:
        部署策略（云端/边缘/混合）
    """
    # 大模型（>30B）：云端部署
    if model_size > 30:
        return "云端"

    # 高延迟要求（<100ms）：边缘部署
    if latency_requirement < 100:
        return "边缘"

    # 高隐私敏感（>=4）：边缘部署
    if privacy_sensitivity >= 4:
        return "边缘"

    # 高并发（>10万请求/天）：云端部署
    if requests_per_day > 100000:
        return "云端"

    # 中等规模：混合部署
    if requests_per_day > 10000:
        return "混合"

    # 小规模：边缘部署
    return "边缘"

# 使用示例
strategy = decide_deployment_strategy(
    model_size=7,
    requests_per_day=50000,
    latency_requirement=300,
    privacy_sensitivity=2
)

print(f"推荐部署策略: {strategy}")
```

```python
# 2. Redis 缓存（减少重复推理）
redis_client = redis.Redis(host='localhost', port=6379)

def cached_generate(prompt, max_tokens=256):
    cache_key = f"gen:{prompt}:{max_tokens}"

    # 检查缓存
    cached_result = redis_client.get(cache_key)
    if cached_result:
        return cached_result.decode()

    # 写入缓存（TTL 1小时）
    redis_client.setex(cache_key, 3600, result)

    return result

# 3. 任务队列（削峰填谷）
queue = Queue(connection=redis_client)

def enqueue_generate_task(prompt, max_tokens):
    job = queue.enqueue(cached_generate, prompt, max_tokens, timeout=30)
    return job.id

# 4. FastAPI 服务
app = FastAPI()

@app.post("/generate")
async def generate_endpoint(prompt: str, max_tokens: int = 256):
    # 提交到队列
    job_id = enqueue_generate_task(prompt, max_tokens)

    # 返回任务 ID
    return {"job_id": job_id}

@app.get("/result/{job_id}")
async def get_result(job_id: str):
    job = queue.fetch_job(job_id)
    if job is None:
        return {"error": "Job not found"}

    if job.is_finished:
        return {"result": job.result, "status": "completed"}
    else:
        return {"status": "pending"}
```

- 硬件：3 x A100（$7.5/小时）
- 月成本：3 x 24 x 30 x $7.5 = $16,200
- 优化后（缓存 + 队列）：月成本降到 $5,000

### 6.2 常见陷阱与解决方案

**陷阱1：过度优化导致精度损失**
- 问题：INT4 量化后精度下降严重
- 解决方案：使用混合量化（关键层 FP16/INT8，非关键层 INT4）

```python
# 混合量化示例
from optimum.intel import OVQuantizer

quantizer = OVQuantizer.from_pretrained(model)
quantizer.quantize(
    calibration_dataset=calibration_data,
    weights_only=True,
    scheme="symmetric",  # 对称量化
    # 仅量化非注意力层
    activations_quantization_params={
        "attention": {"dtype": "fp16"},  # 注意力层 FP16
        "ffn": {"dtype": "int8"}  # FFN 层 INT8
    }
)
```

**陷阱2：内存泄漏导致崩溃**
- 问题：长时间运行后内存占用不断增加
- 解决方案：定期清理 KV Cache 和中间变量

```python
class MemoryOptimizedModel:
    def __init__(self, model):
        self.model = model
        self.cache = None

    def generate(self, input_ids, max_new_tokens=100):
        # 清理旧缓存
        self.clear_cache()

        for _ in range(max_new_tokens):
            with torch.no_grad():
                outputs = self.model(input_ids, past_key_values=self.cache, use_cache=True)

            logits = outputs.logits[:, -1, :]
            next_token = torch.argmax(logits, dim=-1).unsqueeze(1)
            input_ids = torch.cat([input_ids, next_token], dim=-1)
            self.cache = outputs.past_key_values

            if next_token.item() == self.tokenizer.eos_token_id:
                break

        return input_ids

    def clear_cache(self):
        if self.cache is not None:
            # 手动释放缓存
            for layer_cache in self.cache:
                layer_cache[0].storage().resize_(0)  # key
                layer_cache[1].storage().resize_(0)  # value
            self.cache = None
```

**陷阱3：边缘设备硬件兼容性问题**
- 问题：模型在某些设备上无法运行
- 解决方案：提供多版本模型（FP16/INT8/INT4）+ 自动检测

```python
def load_optimal_model():
    """
    根据设备能力加载最优模型版本
    """
    device_info = detect_device_capabilities()

    # 高性能设备（A100/H100）：FP16
    if device_info["gpu_memory"] >= 80e9:
        model = load_model("model_fp16.pt")

    # 中端设备（RTX 4090）：INT8
    elif device_info["gpu_memory"] >= 24e9:
        model = load_model("model_int8.pt")

    # 低端设备（无 GPU）：INT4 + CPU
    elif device_info["gpu_memory"] is None:
        model = load_model("model_int4.pt", device="cpu")

    else:
        raise RuntimeError("Device not supported")

    return model

def detect_device_capabilities():
    """检测设备能力"""
    if torch.cuda.is_available():
        return {
            "gpu_memory": torch.cuda.get_device_properties(0).total_memory,
            "gpu_name": torch.cuda.get_device_name(0)
        }
    else:
        return {
            "gpu_memory": None,
            "gpu_name": None
        }
```
