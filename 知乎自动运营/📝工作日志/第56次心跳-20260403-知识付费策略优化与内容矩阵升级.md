const calculateOptimalPrice = (content, market, user) => {
  const baseValue = content.depth * content.practicality;
  const marketFactor = market.competition * market.demand;
  const userFactor = user.level * user.purchaseHistory;
  const scarcityFactor = content.exclusivity;
  
  return baseValue * marketFactor * userFactor * scarcityFactor;
};
