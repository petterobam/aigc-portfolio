### 技术方案

**提取 Cookie 的代码**:
```javascript
mcporter call playwright.browser_run_code code='async (page) => {
  // 导航到知乎
  await page.goto("https://www.zhihu.com", { waitUntil: "domcontentloaded" });
  await page.waitForTimeout(2000);

  // 获取所有 Cookie
  const context = page.context();
  const cookies = await context.cookies();

  // 筛选出知乎相关的 Cookie
  const zhihuCookies = cookies.filter(c =>
    c.domain.includes("zhihu.com") || c.domain.endsWith(".zhihu.com")
  );

  // 检查登录态关键 Cookie
  const sessionCookieNames = ["z_c0", "d_c0", "q_c1", "zse93"];
  const sessionValid = sessionCookieNames.some(name =>
    zhihuCookies.some(c => c.name === name && c.value)
  );

  // 获取关键 cookie 的过期时间
  const zc0 = zhihuCookies.find(c => c.name === "z_c0");
  const exp = zc0?.expires ? new Date(zc0.expires * 1000).toLocaleString("zh-CN") : "未知";

  return {
    url: page.url(),
    title: await page.title(),
    totalCookies: cookies.length,
    zhihuCookies: zhihuCookies.length,
    httpOnlyCount: zhihuCookies.filter(c => c.httpOnly).length,
    sessionValid,
    zc0Expires: exp,
    cookies: zhihuCookies
  };
}'
```
