```javascript
const answerInputSelectors = [
  '[placeholder="写下你的回答..."]',
  '[contenteditable="true"]',
  '.Public-DraftEditor-content',
  '.DraftEditor-editorContainer .public-DraftEditor-content'
];

let answerInput = null;
for (const selector of answerInputSelectors) {
  answerInput = await page.$(selector);
  if (answerInput) {
    break;
  }
}
```

### 2. HTML 内容填充

**发现**：知乎的富文本编辑器支持 HTML 内容，可以直接使用 `innerHTML` 填充。

**优势**：
- 支持 Markdown 渲染后的 HTML
- 支持代码高亮
- 支持富文本格式

**代码示例**：
```javascript
await page.evaluate((content) => {
  const editor = document.querySelector('.public-DraftEditor-content');
  if (editor) {
    editor.innerHTML = content;
    // 触发 input 和 change 事件
    editor.dispatchEvent(new Event('input', { bubbles: true }));
    editor.dispatchEvent(new Event('change', { bubbles: true }));
  }
}, answerData.content);
```

---

### 3. 人类化操作的重要性

**发现**：模拟真实用户行为可以降低被反爬机制识别的风险。

**关键技巧**：
- 随机延迟（500-2000ms）
- 逐字输入（每 10 个字符暂停）
- 滚动页面（模拟浏览）
- 点击后延迟（等待页面响应）

**代码示例**：
```javascript
async function humanType(page, selector, text, options = {}) {
  const element = await page.$(selector);
  await element.click();
  await randomDelay(CONFIG.delays.action);

  for (let i = 0; i < text.length; i++) {
    await page.keyboard.type(text[i]);
    if (i % 10 === 0) {
      await randomDelay(CONFIG.delays.typing, CONFIG.delays.typing * 2);
    }
  }
}
```
